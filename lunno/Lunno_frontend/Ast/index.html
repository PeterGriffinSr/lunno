type ty =
  1. | TyInt
    (*

    Integer type.

    *)
  2. | TyFloat
    (*

    Floating-point type.

    *)
  3. | TyString
    (*

    String type.

    *)
  4. | TyBool
    (*

    Boolean type.

    *)
  5. | TyUnit
    (*

    Unit type.

    *)
  6. | TyVar of string
    (*

    Type variable, e.g., 'a.

    *)
  7. | TyList of ty
    (*

    List type, e.g., int.

    *)
  8. | TyFunction of ty list * ty
    (*

    Function type, e.g., (int, int) -> int.

    *)

The ty type represents types in the language.

type param = {
  1. name : string;
    (*

    Parameter name.

    *)
  2. ty : ty option;
    (*

    Optional type annotation.

    *)
  3. span : Lunno_common.Span.t;
    (*

    Source code span of the parameter.

    *)
}

The param type represents a function parameter.

type expr =
  1. | IntLiteral of int64 * Lunno_common.Span.t
    (*

    Integer literal.

    *)
  2. | FloatLiteral of float * Lunno_common.Span.t
    (*

    Floating-point literal.

    *)
  3. | StringLiteral of string * Lunno_common.Span.t
    (*

    String literal.

    *)
  4. | BooleanLiteral of bool * Lunno_common.Span.t
    (*

    Boolean literal.

    *)
  5. | UnitLiteral of Lunno_common.Span.t
    (*

    Unit literal.

    *)
  6. | Variable of string * Lunno_common.Span.t
    (*

    Variable reference.

    *)
  7. | Lambda of lambda
    (*

    Lambda / anonymous function.

    *)
  8. | Apply of expr * expr list * Lunno_common.Span.t
    (*

    Function application.

    *)
  9. | Let of let_expr
    (*

    Let binding expression.

    *)
  10. | If of if_expr
    (*

    If expression.

    *)
  11. | Match of match_expr
    (*

    Match expression.

    *)
  12. | Block of expr list * Lunno_common.Span.t
    (*

    Block expression.

    *)
  13. | Binary of binary_expr
    (*

    Binary operation.

    *)
  14. | Unary of unary_expr
    (*

    Unary operation.

    *)

The expr type represents all expressions in the language.

and lambda = {

    (*

    Parameters.

    *)

    (*

    Function body.

    (*

    Span of the whole lambda.

    *)
